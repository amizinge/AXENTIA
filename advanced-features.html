Code Examples

Python SDK Example
import axentia

# Initialize client
client = axentia.Client(api_key='your_api_key')

# Get market data
ticker = client.market.get_ticker('BTC/USD')
print(f"Current Price: ${ticker.price}")

# Place order
order = client.orders.create(
    symbol='BTC/USD',
    side='buy',
    amount=0.1,
    order_type='market'
)

print(f"Order ID: {order.id}")
print(f"Status: {order.status}")
JavaScript SDK Example
const Axentia = require('axentia-sdk');

// Initialize client
const client = new Axentia.Client({
  apiKey: 'your_api_key'
});

// Subscribe to price updates
client.ws.subscribe('BTC/USD', (data) => {
  console.log(`Price: $${data.price}`);
});

// Place automated order
const order = await client.orders.create({
  symbol: 'ETH/USD',
  side: 'sell',
  amount: 0.5,
  type: 'limit',
  price: 2700.00
});

console.log('Order placed:', order.id);
